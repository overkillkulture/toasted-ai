## USAGE GUIDE

### Quick Start

```python
# 1. System Summation
from system_summation_operator import SystemSummationOperator, ComponentType

summation = SystemSummationOperator()
summation.register_component("comp_id", ComponentType.CORE_SYSTEM, "path.py")
result = summation.compute_total_summation()
summation.save_summation_report()

# 2. Integration Automation
from integration_automation_operator import IntegrationAutomationOperator

integration = IntegrationAutomationOperator()
integration.register_component("comp_id", "path.py", dependencies=["dep1"])
integration.integrate_all()
integration.save_integration_report()

# 3. Reconstruction
from reconstruction_optimizer import CompleteReconstructionOptimizer, ReconstructionLayer

reconstruction = CompleteReconstructionOptimizer()
reconstruction.create_blueprint("id", ReconstructionLayer.LAYER_CORE, "path.py")
reconstruction.reconstruct_all()
reconstruction.save_reconstruction_report()

# 4. Decision Tracking
from deterministic_decision_tracker import DeterministicDecisionTracker, DecisionType

tracker = DeterministicDecisionTracker()
state = tracker.capture_system_state()
tracker.record_decision(DecisionType.SYSTEM_ACTION, input_data, output_data, reasoning, state)
tracker.save_decision_report()
```
